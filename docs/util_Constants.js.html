 * @typedef {string} WTEndPoint
 */
exports.EndPoints = keyMirror([
  "main",
  "user",
  "clan",
  "news",
  "changelog"
]);

function keyMirror(arr) {
  const tmp = Object.create(null);
  for (const value of arr) tmp[value] = value;
  return tmp;
}
